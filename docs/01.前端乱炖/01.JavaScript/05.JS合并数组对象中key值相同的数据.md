## 将数组里某个属性相同的对象合并到一个数组中

_原数组示例：_

```javascript
  let oldData = [
    {
      type:"篮球",
      value:20
    },
    {
      type:"跑步",
      value:40
    },
    {
      type:"游泳",
      value:10
    }
    {
      type:"跑步",
      value:15
    }
  ]
```

## 实现方案

### 方法一

```javascript
let tempArr = [];
let data = [];
for (let i = 0; i < oldData.length; i++) {
  if (tempArr.indexOf(oldData[i].type) === -1) {
    data.push({
      type: oldData[i].type,
      infoList: [oldData[i]],
    });
    tempArr.push(oldData[i].type);
  } else {
    for (let j = 0; j < data.length; j++) {
      if (data[j].type == oldData[i].type) {
        data[j].infoList.push(oldData[i]);
        break;
      }
    }
  }
}
```

### 方法二

```javascript
let infoList = {};
oldData.forEach((item, index) => {
  let { type } = item;
  if (!infoList[type]) {
    infoList[type] = {
      type,
      list: [],
    };
  }
  infoList[type].list.push(item);
});

let list = Object.values(infoList); // 转换后的数据
```
